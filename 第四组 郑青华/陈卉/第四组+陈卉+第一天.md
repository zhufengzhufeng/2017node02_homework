### callback.js
```
//如何解决异步问题？ 回调函数
//配置es6              +    配置node环境
//language-> javascript     npm->enable
function read(callback) {
    setTimeout(function () {
       let result = 'hello';
       callback(result);//调用回调函数 将结果传入
    },2000);
}
//当我们调用read方法时传入后续的逻辑的函数，作为参数
read(function (result) {
    console.log(result);
});
```
